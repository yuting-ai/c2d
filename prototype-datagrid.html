#1 💬 Show me revenue by region 1 query ▶
revenue by region SELECT 
SELECT region,
  SUM(revenue) AS total_revenue,
  COUNT(*) AS order_count
FROM sales_2024
GROUP BY region
ORDER BY total_revenue DESC;
#2 💬 What's the monthly revenue trend? 1 query ▶
monthly revenue SELECT 
SELECT DATE_TRUNC('month', date) AS month,
  SUM(revenue) AS revenue
FROM sales_2024
GROUP BY 1
ORDER BY 1;
#3 💬 Investigate revenue outliers above 50K 2 queries ▶
compute z-scores STATS 
WITH stats AS (
  SELECT AVG(revenue) AS mu,
    STDDEV(revenue) AS sigma
  FROM sales_2024
)
SELECT *, (revenue - mu) / sigma AS z_score
FROM sales_2024, stats
ORDER BY z_score DESC;
filter outliers SELECT 
SELECT * FROM sales_2024, stats
WHERE ABS(revenue - mu) / sigma > 3
ORDER BY revenue DESC;
